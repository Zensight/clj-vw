Set one or more options. Can be chained, e.g.

(def settings (-> {}
             (set-option :data "foo/bar.dat")
             (set-option :save-resume true)
             (set-option :ngram 3)
             (set-option :quadratic "ab")
             (set-option :quadratic "ac")
             (set-option :learning-rate 0.3)))
